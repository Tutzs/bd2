ACESSAR O MYSQL

mysql -u root -p                              -- conecta ao MySQL solicitando senha de forma interativa
mysql -u root -p nome_do_banco                -- conecta e já seleciona um banco de dados específico
QUIT;                                         -- sai do MySQL (mesma função do EXIT)

------------------------------------------------------------------
BACKUP LOGICO (via mysqldump - executado no terminal, fora do MySQL)
------------------------------------------------------------------

mysqldump -u root -p nome_do_banco > backup.sql      -- gera backup logico completo do banco bd_empresa
mysql -u root -p nome_do_banco < backup.sql          -- restaura um backup logico dentro do banco bd_empresa

Precisa ter o banco criado para restaurar o backup

------------------------------------------------------------------
NAVEGACAO E INFORMACOES GERAIS
------------------------------------------------------------------

SHOW DATABASES;                                   -- lista todos os bancos de dados existentes
USE nome_do_banco;                                -- seleciona um banco de dados para trabalhar
SHOW TABLES;                                      -- lista as tabelas do banco de dados selecionado
DESC nome_tabela;                                 -- mesma funcao do DESCRIBE (forma abreviada)
SELECT DATABASE();                                -- mostra qual banco de dados esta em uso no momento
STATUS;                                           -- mostra informacoes da conexao e do servidor atual
SHOW CREATE TABLE nome_tabela;                    -- mostra o comando CREATE TABLE completo da tabela

Uso "Show create table" para ver como a tabela foi criada, caso precise criar uma auxiliar com os mesmos atributos e regras de negócio.

------------------------------------------------------------------
DDL - DATA DEFINITION LANGUAGE (definicao de estrutura)
------------------------------------------------------------------

CREATE DATABASE IF NOT EXISTS nome_do_banco;      -- cria um banco de dados, se ele ainda nao existir
DROP DATABASE nome_do_banco;                      -- apaga um banco de dados por completo

CREATE TABLE nome_tabela (                        -- cria uma nova tabela
    id INT AUTO_INCREMENT PRIMARY KEY,            -- coluna id como chave primaria autoincrementada
    nome VARCHAR(100) NOT NULL,                   -- coluna nome, texto obrigatorio (nao nulo)
    email VARCHAR(100) UNIQUE,                    -- constraint UNIQUE: nao permite valores repetidos
    idade INT CHECK (idade >= 0),                 -- constraint CHECK: valida uma condicao para a coluna
    id_outra_tabela INT,                          -- coluna que vai servir de chave estrangeira (FK)
    CONSTRAINT fk_nome_outra_tabela                -- nome da constraint (facilita alterar/remover depois)
        FOREIGN KEY (id_outra_tabela)              -- FK: liga essa coluna a outra tabela
        REFERENCES outra_tabela(id)                -- tabela e coluna referenciadas (geralmente a PK)
        ON DELETE CASCADE                          -- se o registro pai for apagado, apaga os filhos tambem
        ON UPDATE CASCADE                          -- se o id pai mudar, atualiza automaticamente aqui
);

ALTER TABLE nome_tabela                           -- adiciona uma FK em uma tabela ja existente
    ADD CONSTRAINT fk_nome_outra_tabela
    FOREIGN KEY (id_outra_tabela) REFERENCES outra_tabela(id);

ALTER TABLE nome_tabela DROP FOREIGN KEY fk_nome_outra_tabela; -- remove uma foreign key pelo nome da constraint

ALTER TABLE nome_tabela ADD COLUNA VARCHAR(50);   -- adiciona uma nova coluna a uma tabela existente
ALTER TABLE nome_tabela MODIFY COLUNA INT;        -- altera o tipo de dado de uma coluna existente
ALTER TABLE nome_tabela DROP COLUMN nome_coluna;  -- remove uma coluna de uma tabela existente

DROP TABLE nome_tabela;                           -- apaga a tabela e todos os seus dados permanentemente
TRUNCATE TABLE nome_tabela;                       -- apaga todos os dados da tabela, mas mantem a estrutura

------------------------------------------------------------------
DML - DATA MANIPULATION LANGUAGE (manipulacao de dados)
------------------------------------------------------------------

SELECT * FROM nome_tabela;                        -- seleciona e exibe todos os registros da tabela
SELECT coluna1, coluna2 FROM nome_tabela           -- seleciona colunas especificas
WHERE condicao;                                   -- filtra os registros de acordo com uma condicao

INSERT INTO nome_tabela (coluna1, coluna2)         -- insere um novo registro na tabela
VALUES ('valor1', 'valor2');

UPDATE nome_tabela SET coluna1 = 'novo_valor'      -- atualiza valores de registros existentes
WHERE condicao;                                   -- (sempre use WHERE, senao atualiza a tabela inteira)

DELETE FROM nome_tabela WHERE condicao;           -- exclui registros que atendem a condicao especificada

------------------------------------------------------------------
VIEWS (consultas salvas, funcionam como tabelas virtuais)
------------------------------------------------------------------

CREATE VIEW nome_view AS                          -- cria uma view a partir de uma consulta
    SELECT coluna1, coluna2 FROM nome_tabela
    WHERE condicao;

CREATE OR REPLACE VIEW nome_view AS               -- cria a view ou substitui se ja existir
    SELECT coluna1, coluna2 FROM nome_tabela
    WHERE condicao;

SELECT * FROM nome_view;                          -- consulta a view como se fosse uma tabela normal
SHOW CREATE VIEW nome_view;                        -- mostra o comando usado para criar a view
DROP VIEW nome_view;                               -- apaga a view (nao afeta os dados das tabelas originais)

------------------------------------------------------------------
OTIMIZACAO E DIAGNOSTICO (uteis para tuning de consultas)
------------------------------------------------------------------

EXPLAIN SELECT * FROM tabela WHERE coluna = valor; -- mostra o plano de execucao usado pelo SGBD na consulta

CREATE INDEX nome_indice ON nome_tabela (coluna);  -- cria um indice em uma coluna para acelerar buscas/joins
CREATE UNIQUE INDEX nome_indice ON nome_tabela (coluna); -- cria um indice que tambem garante valores unicos
DROP INDEX nome_indice ON nome_tabela;             -- remove um indice existente da tabela

SHOW INDEX FROM nome_tabela;                       -- lista os indices existentes em uma tabela
SHOW PROCESSLIST;                                  -- mostra os processos/consultas em execucao no momento
SHOW ENGINE INNODB STATUS;                         -- mostra status detalhado do motor de armazenamento InnoDB

Para executar algum script direto de fora

mysql -u root -p nome_do_banco < script.sql    -- executa um arquivo .sql inteiro no banco especificado
